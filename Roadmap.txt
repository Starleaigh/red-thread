// items/skills.js
export function skillValue(item) {
  const base = item.system.base ?? 0;
  const occupation = item.system.occupation ? 20 : 0;
  return base + occupation;
}

export function skillHalf(item) {
  return Math.floor(skillValue(item) / 2);
}

// items/weapons.js
export async function rollWeaponDamage(item) {
  if (!item.isWeapon) return null;
  return new Roll(item.system.damage).roll({async: true});
}

//red-thread-tiem.js

import * as Skills from "./skills.js";
import * as Weapons from "./weapons.js";

export class RedThreadItem extends Item {

  get isSkill() { return this.type === "skill"; }
  get isWeapon() { return this.type === "weapon"; }

  get value() {
    if (this.isSkill) return Skills.skillValue(this);
    return null;
  }

  get half() {
    if (this.isSkill) return Skills.skillHalf(this);
    return null;
  }

  async rollDamage() {
    if (this.isWeapon) return Weapons.rollWeaponDamage(this);
  }
}

  if (!skillId) return;

  }

Actor data → Build STATE object → Diff → Patch DOM

state = {
  actor: null,
  skills: [],
  system: null
};

_buildState() {
  const actor = this.actor.isToken
    ? this.actor.baseActor
    : this.actor;

  const skills = actor.items
    .filter(i => i.type === "skill")
    .map(skill => ({
      id: skill.id,
      name: skill.name,
      base: skill.system.base ?? 0,
      value: skill.value,
      half: skill.half,
      fifth: skill.fifth
    }));

  return {
    actor,
    system: actor.system,
    skills
  };
}

_onRender(){
this.state = this._buildState();
this._renderState();

}

_renderState() {
  const skillContainer = this.element.querySelector(".skill-container");
  if (!skillContainer) return;

  const existingRows = new Map(
    [...skillContainer.querySelectorAll("[data-item-id]")]
      .map(el => [el.dataset.itemId, el])
  );

  for (const skill of this.state.skills) {

    let row = existingRows.get(skill.id);

    if (!row) {
      // Create new row
      row = document.createElement("div");
      row.dataset.itemId = skill.id;
      row.classList.add("skill-row");

      row.innerHTML = `
        <label>${skill.name}</label>
        <input type="number" data-field="system.base" value="${skill.base}" />
        <span class="skill-half">${skill.half}</span>
        <span class="skill-fifth">${skill.fifth}</span>
      `;

      skillContainer.appendChild(row);
    }

    // Update existing row
    row.querySelector(".skill-half").textContent = skill.half;
    row.querySelector(".skill-fifth").textContent = skill.fifth;
  }

  // Remove deleted skills
  for (const [id, row] of existingRows) {
    if (!this.state.skills.find(s => s.id === id)) {
      row.remove();
    }
  }
}

async _onUpdateEmbeddedDocuments(parent, collection, docs, changes, options) {
  if (collection === "items") {
    this.state = this._buildState();
    this._renderState();
  }

  return super._onUpdateEmbeddedDocuments(parent, collection, docs, changes, options);
}
